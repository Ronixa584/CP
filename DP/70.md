class Solution {
public:
    int climbStairs(int n) {
        if(n==1) return 1;
        if(n==2) return 2;
        if(n==3) return 3;
        return climbStairs(n-1)+climbStairs(n-2);
    }
};

class Solution {
public:

    int help(int n,vector<int>&dp){
        if(n==2) return 2;
        if(n==3) return 3;

        if(dp[n]!=-1) return dp[n];

        dp[n]=help(n-1,dp)+help(n-2,dp);

        return dp[n];
    }
    
    int climbStairs(int n) {
        if(n<=3) return n;
        vector<int>dp(n+1,-1);
        return help(n,dp);
    }
};
```

class Solution {
public:
    int climbStairs(int n) {
        if(n==1 || n==2){
            return n;
        }
        int p1=2;
        int p2=1;
        int c=0;
        for(int i=3;i<=n;i++){
            c=p1+p2;
            p2=p1;
            p1=c;
        }
        return c;
    }
};
```
